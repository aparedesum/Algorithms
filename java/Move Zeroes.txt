public class Solution {
    public void moveZeroes(int[] nums) {
        int n = nums.length;
        int i=0;
        int firstZero=0;
        
        while(i<n){
            if(nums[i]!=0){
                int tmp = nums[i];
                nums[i]=nums[firstZero];
                nums[firstZero]= tmp;
                firstZero++;
            }
            i++;
        }
    }
}
