//https://leetcode.com/problems/assign-cookies/#/description

public class Solution {
    public int findContentChildren(int[] g, int[] s) {
         Arrays.sort(g);
		 Arrays.sort(s);
		 
		 int n = g.length;
		 int m = s.length;
		 int i = 0;
		 int j = 0;
		 int count = 0;
		 
		 while(i<n && j <m ){
			 int son = g[i];
			 int cookie = s[j++];
			 
			 if(son <= cookie){
				 count++;
				 i++;				 
			 } 			 	
		 }		 
		 return count;
    }
}
